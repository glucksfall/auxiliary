Unin del inductor $S_x$ a su ligando $X$

$$X + S_x \leftrightharpoons XS_x$$¶

Ecuacion de Michaelis Menten:

$$ XS_x = \frac{X_T S_x}{S_x + K_x} $$¶

Ecuacion de Hill:

$$ nXS_x = \frac{X_T S_x^n}{S_x^n + K_x^n} $$¶

def michaelis(init, t, par):
    Sx = init[0]
    XT = par[0] # los elementos en una lista se acceden usando [] (en MATLAB usando parentesis)
    Kx = par[1] # el indice empieza en 0 (en MATLAB empieza en 1)
    
    # definicion de las ecuaciones
    XSx = XT * Sx / (Sx + Kx)
    
    # python es similar a C, donde cada funcin debe retornar un resultado
    return [-XSx, -XSx, +XSx]

def hill(init, t, par):
    Sx = init[0]
    XT = par[0]
    Kx = par[1]
    n = par[2]
    
    nXSx = XT * Sx**n / (Sx**n + Kx**n)
    
    return [-nXSx, -nXSx, +nXSx]

def xy(t, init):
    bx = 10
    ax = 0.1
    kxx = 80
    kxy = 40
    
    by = 10
    ay = 0.1
    kyy = 80
    
    x = init[0]
    y = init[1]

    dx = bx * int(x <= kxx) - ax * x
    dy = by * int(x <= kxy) * int(y < kyy) - ay * y
    
    return [dx, dy]

def xyz(x, t):
    sx = 0
    if (t > 25):
        sx = 50
    ksx = 25
    
    bx = 10
    ax = 0.1
    kxy = 90
    
    by = 10
    ay = 0.1
    kyz = 90
    
    bz = 10
    az = 0.1
    
    dx = bx - ax * x[0]
    x_act = x[0] * int(sx > ksx)
    dy = by * int(x_act > kxy) - ay * x[1]

    ksy = 50
    
    bx = 10
    ax = 0.1
    by = 10
    bybasal = 0.1
    ay = 0.1
    bz = 10
    bzbasal = 5
    az = 0.1
    
    kxy = 90
    kxz = 75
    kyz = 75
    kyy = 75
    
    dx = 0
    y_act = x[1] * int(sy < ksy) # x[1] si sy = 0
    dy = by * int(x[0] > kxy) - bybasal * int(y_act < kyy) - ay * x[1]
